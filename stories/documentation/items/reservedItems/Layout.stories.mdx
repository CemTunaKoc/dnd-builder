# Layout Item

This item is reserved for the general layout settings.

its item type is  `__layout__`. and you should not use this keyword for other items.

<Canvas>
    <BuilderProvider>
      {(() => {
        const [_layout, editLayout] = useState({
          reportBackgroundColor: '#fa8900',
          reportLayout: 'Web',
          reportLayoutHeight: '800',
          reportLayoutWidth: '800',
        });
        return (
          <div  style={{ display: 'block', height: '350px' }}>
            <div className="jfReport">
              <PropProvider settings={_layout}>
                <BuilderProvider isRightPanelOpen>
                  <ReportWrapper>
                    <RightPanel onSettingChange={(id, setting) => {editLayout({ ..._layout, ...setting})}} />
                  </ReportWrapper>
                </BuilderProvider>
              </PropProvider>
            </div>
          </div>
        )
      })()}
    </BuilderProvider>
</Canvas>
